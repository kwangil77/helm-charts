1. Get the application URL by running these commands:
{{- if .Values.controller.ingress.enabled }}
{{- range .Values.controller.ingress.hosts }}
  http{{ if $.Values.controller.ingress.tls }}s{{ end }}://{{ . }}/
{{- end }}
{{- else }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "boundary.name" . }},app.kubernetes.io/instance={{ .Release.Name }},app.kubernetes.io/component=controller" -o jsonpath="{.items[0].metadata.name}")
  export CONTAINER_PORT=$(kubectl get pod --namespace {{ .Release.Namespace }} $POD_NAME -o jsonpath="{.spec.containers[0].ports[0].containerPort}")
  echo "Visit http://127.0.0.1 to use your application"
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 80:$CONTAINER_PORT
{{- end }}
